#include <bits/stdc++.h>
using namespace std;
struct Employee {
    string name;            
    double hourlySalary;    
    double taxWithholdings; 
};
double calculateNetSalary(double hourlySalary, double taxWithholdings) {
    const double hoursWorkedPerWeek = 40; 
    double grossSalary = hoursWorkedPerWeek * hourlySalary * 4;  
    double taxes = grossSalary * (taxWithholdings / 100);  
    double netSalary = grossSalary - taxes;  
    return netSalary;
}
int main() {
    vector<Employee> employees;
    employees.push_back({"Jones", 20.0, 10.0});
    employees.push_back({"Artem", 25.0, 15.0});
    cout << "Employee Information:" << endl;
    for (const auto& emp : employees) {
        cout<<"Name: " << emp.name << endl;
        cout<<"Hourly Salary: $" << emp.hourlySalary<<endl;
        cout<<"Tax Withholdings (%): "<<emp.taxWithholdings << endl;
        cout<<"Net Salary: $" <<calculateNetSalary(emp.hourlySalary, emp.taxWithholdings) << endl;
        cout << endl;  
    }

    return 0;
}
