#include <iostream>
#include <cstdlib>
#include <ctime>

const int NUM_CARDS = 20;
int cards[NUM_CARDS];

void initializeCards() {
    for (int i = 0; i < NUM_CARDS; ++i) {
        cards[i] = i % (NUM_CARDS / 2) + 1; // Generating pairs of numbers
    }
    // Shuffle the cards
    srand(time(NULL));
    for (int i = 0; i < NUM_CARDS; ++i) {
        int randomIndex = rand() % NUM_CARDS;
        int temp = cards[i];
        cards[i] = cards[randomIndex];
        cards[randomIndex] = temp;
    }
}

void displayCards() {
    std::cout << "Current Board:" << std::endl;
    for (int i = 0; i < NUM_CARDS; ++i) {
        if (i % 5 == 0) {
            std::cout << std::endl;
        }
        if (cards[i] == -1) {
            std::cout << "X ";
        } else {
            std::cout << cards[i] << " ";
        }
    }
    std::cout << std::endl << std::endl;
}

bool allMatched() {
    for (int i = 0; i < NUM_CARDS; ++i) {
        if (cards[i] != -1) {
            return false;
        }
    }
    return true;
}

void game() {
    initializeCards();
    int moves = 0;
    while (!allMatched()) {
        displayCards();
        int index1, index2;
        std::cout << "Enter index of first card: ";
        std::cin >> index1;
        std::cout << "Enter index of second card: ";
        std::cin >> index2;

        if (index1 == index2 || index1 < 0 || index2 < 0 || index1 >= NUM_CARDS || index2 >= NUM_CARDS) {
            std::cout << "Invalid indices. Please enter valid indices." << std::endl;
            continue;
        }

        if (cards[index1] == cards[index2]) {
            std::cout << "Match found!" << std::endl;
            cards[index1] = cards[index2] = -1; // Keeping cards face up
        } else {
            std::cout << "No match. Try again." << std::endl;
        }
        moves++;
    }

    std::cout << "Congratulations! All pairs matched." << std::endl;
    std::cout << "Total moves taken: " << moves << std::endl;
}

int main() {
    game();
    return 0;
}
